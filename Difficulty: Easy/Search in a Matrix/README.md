<pre><span style="font-size: 14pt;"><strong>Input</strong>: mat[][] = [[6, 23, 21],[4, 45, 32],[69, 11, 87]], x = 32

</span></pre>
<pre><span style="font-size: 14pt;"><strong>Input</strong>: mat[][] = [[14, 34, 23, 95, 43, 28]], x = 55<br><strong>Output</strong>: false
